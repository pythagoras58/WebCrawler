

    function rgb2hex(orig) {
        var rgb = orig.replace(/\s/g, "").match(/^rgba?\((\d+),(\d+),(\d+)/i);
        return rgb && rgb.length === 4
            ? "#" +
            ("0" + parseInt(rgb[1], 10).toString(16)).slice(-2) +
            ("0" + parseInt(rgb[2], 10).toString(16)).slice(-2) +
            ("0" + parseInt(rgb[3], 10).toString(16)).slice(-2)
            : orig;
    }

    function lightOrDark(color) {
        // Variables for red, green, blue values
        var r, g, b, hsp;

        // Check the format of the color, HEX or RGB?
        if (color.match(/^rgb/)) {
            // If RGB --> store the red, green, blue values in separate variables
            color = color.match(
                /^rgba?\((\d+),\s*(\d+),\s*(\d+)(?:,\s*(\d+(?:\.\d+)?))?\)$/
            );

            r = color[1];
            g = color[2];
            b = color[3];
        } else {
            // If hex --> Convert it to RGB: http://gist.github.com/983661
            color = +(
                "0x" + color.slice(1).replace(color.length < 5 && /./g, "$&$&")
            );

            r = color >> 16;
            g = (color >> 8) & 255;
            b = color & 255;
        }

        // HSP (Highly Sensitive Poo) equation from http://alienryderflex.com/hsp.html
        hsp = Math.sqrt(0.299 * (r * r) + 0.587 * (g * g) + 0.114 * (b * b));

        // Using the HSP value, determine whether the color is light or dark
        if (hsp > 127.5) {
            return "light";
        } else {
            return "dark";
        }
    }

            [...allBlocks].forEach((block) => {

                                    }

                                }, 90);
                            }
                        });

                        appHeight += block.offsetHeight;
                    }
                }
                stickFooterToTheBottom();
            });

        function stickFooterToTheBottom() {
            // if not all blocks size is calculated, wait
            if (blocksCountToProcess !== 0) {
                return;
            }

            // in case of method works after resize, we need to remove old DIV
            let extraDiv = document.querySelector('.extra-div');
            if (extraDiv) {
                extraDiv.remove();
            }

            // if app height is bigger then window, no need to push footer down, as it's already there
            if (appHeight > window.innerHeight) {
                return;
            }

            if(last){
                let dataBlockAttribute = last.getAttribute('data-block');
                if(!dataBlockAttribute) {
                    return;
                }
                let lastBlockType = dataBlockAttribute.slice(0, 6);
                // if there no footer, nothing to do
                if (lastBlockType !== 'footer') {
                    return;
                }
            }

            const divEl = document.createElement("div");
            divEl.setAttribute("class", "extra-div");
            if(prelast.firstChild){
                divEl.style.background = window.getComputedStyle(isReactBlock(prelast) ? prelast.firstChild : prelast).getPropertyValue("background-color");
            }
            // calculate size for dummy div
            divEl.style.height = (window.innerHeight - appHeight) + 'px';
            divEl.style.width = '100%';

            document.querySelector('[data-appid]').insertBefore(divEl, last);
        }
